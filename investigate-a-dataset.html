def count_TIL(fz,fm):
    #count the totoal number of patients groupby time interval level(TIL)
    total_z=fz.groupby('Time_interval_levels').count().No_show    
    total_m=fm.groupby('Time_interval_levels').count().No_show
    prop_fnm=total_z/total_m
    return prop_fnm
